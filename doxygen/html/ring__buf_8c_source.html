35 if (!out || !cap || !slot_size)

87bool ring_buf_push(ring_buf_t *rb, const ring_slot_hdr_t *hdr, const uint8_t *payload)
88{
89 if (!rb || !hdr || !payload) return false;
90 bool ok=false;
91
92 if (xSemaphoreTake(rb->lock, 0) == pdTRUE) {
93 uint32_t next = nxt(rb->head, rb->cap);
94 if (next != rb->tail) {
95 uint8_t *slot = rb->slots + ((size_t)rb->head * rb->slot_size);
96 memcpy(slot, hdr, sizeof(*hdr));
97 memcpy(slot + sizeof(*hdr), payload, hdr->len);
98 rb->head = next;
99 ok=true;
100 }
101 xSemaphoreGive(rb->lock);
102 }
103 return ok;
104}

114bool ring_buf_pop(ring_buf_t *rb, ring_slot_hdr_t *hdr, uint8_t *payload)
115{
116 if (!rb || !hdr || !payload) return false;
117 bool ok=false;
118
119 if (xSemaphoreTake(rb->lock, pdMS_TO_TICKS(1)) == pdTRUE) {
120 if (rb->tail != rb->head) {
121 uint8_t *slot = rb->slots + ((size_t)rb->tail * rb->slot_size);
122 memcpy(hdr, slot, sizeof(*hdr));
123 memcpy(payload, slot + sizeof(*hdr), hdr->len);
124 rb->tail = nxt(rb->tail, rb->cap);
125 ok=true;
126 }
127 xSemaphoreGive(rb->lock);
128 }
