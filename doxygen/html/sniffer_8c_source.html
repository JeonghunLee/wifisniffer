179void sniffer_enable_promiscuous(void)
180{
181 // Setup Wi-Fi promiscuous filter
182 // https://docs.espressif.com/projects/esp-idf/en/release-v5.4/esp32/api-reference/network/esp_wifi.html
183 // https://docs.espressif.com/projects/esp-idf/en/release-v5.4/esp32/api-reference/network/esp_wifi.html#id5
184 wifi_promiscuous_filter_t filt = {
185 .filter_mask = WIFI_PROMIS_FILTER_MASK_MGMT |
186 WIFI_PROMIS_FILTER_MASK_DATA |
187 WIFI_PROMIS_FILTER_MASK_CTRL,
188 };
189
190 esp_err_t err = esp_wifi_set_promiscuous(false);
191 if (err != ESP_OK) {
192 ESP_LOGW(TAG, "set_promiscuous(false) failed: %d", err);
193 }
194
195 // Setup filter and callback
196 ESP_ERROR_CHECK(esp_wifi_set_promiscuous_filter(&filt));
197 ESP_ERROR_CHECK(esp_wifi_set_promiscuous_rx_cb(&wifi_promiscuous_cb));
198 ESP_ERROR_CHECK(esp_wifi_set_promiscuous(true));
199
200 ESP_LOGI(TAG, "Promiscuous mode enabled");
201}
