77esp_err_t wifi_mgr_init(void)
78{
79 if (s_initialized) {

115esp_err_t wifi_mgr_set_mode_str(const char *mode_str)
116{
117 if (mode_str == NULL) {
118 return ESP_ERR_INVALID_ARG;
119 }
120
121 wifi_mode_t mode = WIFI_MODE_NULL;
122
123 if (strcasecmp(mode_str, "null") == 0) {
124 mode = WIFI_MODE_NULL;
125 }
126 else if (strcasecmp(mode_str, "sta") == 0) {
127 mode = WIFI_MODE_STA;
128 }
129 else if (strcasecmp(mode_str, "ap") == 0) {
130 mode = WIFI_MODE_AP;
131 }
132 else if (strcasecmp(mode_str, "ap+sta") == 0 ||
133 strcasecmp(mode_str, "apsta") == 0 ||
134 strcasecmp(mode_str, "sta+ap") == 0) {
135 mode = WIFI_MODE_APSTA;
136 }
137 else {
138 return ESP_ERR_INVALID_ARG;
139 }
140
141 // 모드 적용
142 esp_err_t err = wifi_mgr_set_mode(mode);
143 if (err != ESP_OK) {
144 return err;
145 }
146
147 return ESP_OK;
148}

220esp_err_t wifi_mgr_set_channel(uint8_t primary)
221{
222 wifi_mode_t mode = WIFI_MODE_NULL;
223
224 if (primary < 1 || primary > 13) { // 필요하면 14까지 허용
225 return ESP_ERR_INVALID_ARG;
226 }
227
228 ESP_ERROR_CHECK(wifi_mgr_init());
229 (void)esp_wifi_get_mode(&mode);
230
231 // 1) STA / APSTA인 경우: 경고 또는 막기
232 if (mode == WIFI_MODE_STA || mode == WIFI_MODE_APSTA) {
233 // 여기서 정책 결정:
234 // - 그냥 막을지
235 // - 또는 경고만 찍고 시도할지
236 ESP_LOGW(TAG,
237 "Channel change in STA/APSTA mode may be invalid or break connection. "
238 "Use NULL/AP mode for sniffer channel hopping.");
239 // 나는 아예 막는 쪽이 더 안전하다고 봄
240 return ESP_ERR_INVALID_STATE;
241 }
242 // 2) NULL / AP 모드인 경우: 채널 변경 시도
243 esp_err_t err = esp_wifi_set_channel(primary, WIFI_SECOND_CHAN_NONE);
244 if (err != ESP_OK) {
245 ESP_LOGE(TAG, "esp_wifi_set_channel failed: %s", esp_err_to_name(err));
246 return err;
247 }
248
249 // 채널 변경 후 스니퍼 다시 세팅 (필요시)
250 sniffer_enable_promiscuous();
251
252 ESP_LOGI(TAG, "Wi-Fi channel set to %u", primary);
253 return ESP_OK;
254}

260void wifi_mgr_print_status(void)
261{
262 wifi_mode_t mode = WIFI_MODE_NULL;
263 (void)esp_wifi_get_mode(&mode);
264
265 wifi_config_t sta;
266 wifi_config_t ap;
267 memset(&sta, 0, sizeof(sta));
268 memset(&ap, 0, sizeof(ap));
269
270 if (esp_wifi_get_config(WIFI_IF_STA, &sta) != ESP_OK) {
271 sta = s_sta_config;
272 }
273 if (esp_wifi_get_config(WIFI_IF_AP, &ap) != ESP_OK) {
274 ap = s_ap_config;
275 }
276
277 printf("Wi-Fi Status :\n");
278 printf(" - Wi-Fi MODE: %s\n", wifi_mgr_mode_to_string(mode));
279 printf(" - STA SSID: %s\n", sta.sta.ssid[0] ? (char *)sta.sta.ssid : "<unset>");
280 printf(" - STA password: %s (%s)\n", sta.sta.password[0] ? "<set>" : "<empty>",sta.sta.password);
281 printf(" - AP SSID: %s\n", ap.ap.ssid[0] ? (char *)ap.ap.ssid : "<unset>");
282 printf(" - AP password: %s\n",
283 (ap.ap.authmode == WIFI_AUTH_OPEN) ? "<open>" :
284 (ap.ap.password[0] ? "<set>" : "<invalid>"));
285 printf(" - AP channel: %u\n", ap.ap.channel);

288// Wi-Fi 모드 적용 내부 함수
294static esp_err_t apply_wifi_mode_internal(wifi_mode_t mode)
295{
296 esp_err_t err = esp_wifi_stop();
297 if (err != ESP_OK &&
298 err != ESP_ERR_WIFI_NOT_INIT &&
299 err != ESP_ERR_WIFI_NOT_STARTED) {
300 ESP_LOGE(TAG, "esp_wifi_stop failed: %s", esp_err_to_name(err));
301 return err;
302 }
303
304 err = esp_wifi_set_mode(mode);
305 if (err != ESP_OK) {
306 ESP_LOGE(TAG, "esp_wifi_set_mode failed: %s", esp_err_to_name(err));
307 return err;
308 }
309
310 if (mode == WIFI_MODE_STA || mode == WIFI_MODE_APSTA) {
311 err = esp_wifi_set_config(WIFI_IF_STA, &s_sta_config);
312 if (err != ESP_OK) {
313 ESP_LOGE(TAG, "STA config failed: %s", esp_err_to_name(err));
314 return err;
315 }
316 }
317
318 if (mode == WIFI_MODE_AP || mode == WIFI_MODE_APSTA) {
319 wifi_config_t cfg = s_ap_config;
320 cfg.ap.ssid_len = (uint8_t)strnlen((const char *)cfg.ap.ssid,
321 sizeof(cfg.ap.ssid));
322 size_t pwd_len = strnlen((const char *)cfg.ap.password,
323 sizeof(cfg.ap.password));
324 cfg.ap.authmode = (pwd_len >= 8) ? WIFI_AUTH_WPA2_PSK : WIFI_AUTH_OPEN;
325 if (cfg.ap.authmode == WIFI_AUTH_OPEN) {
326 memset(cfg.ap.password, 0, sizeof(cfg.ap.password));
327 }
328 ensure_ap_defaults();
329 err = esp_wifi_set_config(WIFI_IF_AP, &cfg);
330 if (err != ESP_OK) {
331 ESP_LOGE(TAG, "AP config failed: %s", esp_err_to_name(err));
332 return err;
333 }
334 }
335
336 err = esp_wifi_start();
337 if (err != ESP_OK) {
338 ESP_LOGE(TAG, "esp_wifi_start failed: %s", esp_err_to_name(err));
339 return err;
340 }
341
342 s_current_mode = mode;
343
344 // 여기서 스니퍼용 promiscuous 모드 활성화
345 sniffer_enable_promiscuous();
346
347 ESP_LOGI(TAG, "Wi-Fi mode set to %s", wifi_mgr_mode_to_string(mode));
348 return ESP_OK;
349}

414 return ESP_ERR_INVALID_ARG;

417 memcpy(s_ap_config.ap.password, password, len);
418 s_ap_config.ap.authmode = WIFI_AUTH_WPA2_PSK;
419 }
420 ensure_ap_defaults();
421 wifi_config_t cfg = s_ap_config;
422 cfg.ap.ssid_len = (uint8_t)strnlen((const char *)cfg.ap.ssid,
423 sizeof(cfg.ap.ssid));
424 return esp_wifi_set_config(WIFI_IF_AP, &cfg);
